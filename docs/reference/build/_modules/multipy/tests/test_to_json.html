def test_to_json4():
    truth4  = mp.truth_scope((1, 15), (1, 10))
    matrix4 = mp.Matrix(4)
    alg4    = mp.Algorithm(matrix4)
    filename = 'test_to_json4.json'
    gen = mp.shallow_truth_table(truth4, alg4)
    mp.json_pretty_store(gen, filename)
